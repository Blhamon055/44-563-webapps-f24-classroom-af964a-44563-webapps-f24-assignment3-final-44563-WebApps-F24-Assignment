def merge(*args, missing_val = None):
            max_length = max([len(lst) for lst in args])
            out_list = []
          
            for i in range(max_length):
              out_list.append([args[k][i] if i < len(args[k]) else missing_val for k in range(len(args))])
          
            return out_list
